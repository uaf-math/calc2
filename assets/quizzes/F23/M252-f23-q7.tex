\problem{8} For each series below (i) write the series using $\displaystyle{\sum}$ notation, (ii) determine whether the series converges, (iii) explain your reasoning, (iv) if the series converges, determine its sum.

	\begin{subproblems}
	\item $\displaystyle{2+\frac{2}{\pi} + \frac{2}{\pi^2} + \frac{2}{\pi^3}+\frac{2}{\pi^4} + \cdots}$
	\vfill
	\item $\displaystyle{-\frac{4}{3}+\frac{16}{9}-\frac{64}{27}+\frac{256}{81}-\cdots}$
	\vfill
	\end{subproblems}
\newpage
\problem{3} Given the series $\displaystyle{\sum_{n=1}^\infty \left(\frac{3}{n+3}-\frac{3}{n+4}\right)}.$
\begin{subproblems}
	\item Find $S_k,$ the $k$th partial sum of the series.
	\vfill
	\item Use $S_k$ to determine the value of series or explain why the series diverges.
	\vfill
	\end{subproblems}
\newpage

\problem{4} Use the Integral Test to determine whether the series $\displaystyle{\sum_{n=1}^\infty ne^{-n^2}}$ converges or diverges.
\vfill
\problem{2} State what is meant by the \emph{harmonic series} and whether the series converges or diverges.
\vfill
\newpage
\problem{8} Determine whether the series below converge or diverge. Explain your reasoning.
\begin{subproblems}
	\item $\displaystyle{\sum_{n=1}^\infty \frac{1}{\sqrt[3]{n^4}}}$
	\vfill
	\item $\displaystyle{\sum_{n=1}^\infty \frac{n}{\ln(n)}}$
	\vfill
	\end{subproblems}
