24 points possible; each part is worth 2 points.  No aids (book, notes, calculator, phone, etc.) are permitted.  Show all work and use proper notation for full credit.  Answers should be in reasonably-simplified form.

\problem{12}  Compute the derivatives of the following functions.

% problems from  Fall 2022 Derivative-v3.pdf 
\begin{subproblems}
\item  $\ds f(x)=\frac{\sqrt{x}}{3}+\frac{5}{\sqrt{x}}-\frac{\sqrt{\pi}}{3}$
\vfill

\item   $\ds f(x) =(\cos(4x)+e^x)^3$
\vfill

\item   $\ds h(x)=\ln(a+x^b)$ where $a$ and $b$ are constants
\vfill

\newpage
\item   $\ds f(x) =\sec(x)\tan(x)$
\vfill


\item  $\ds h(\theta) = \frac{\sin(\theta)}{e^{2\theta}}$
\vfill

\item  Find $\ds \frac{dy}{dx}$ if $\ds e^y +x^3=10+xy$.  You must solve for $\ds \frac{dy}{dx}$.
\vfill
\end{subproblems}


\newpage
\problem{12}  Compute the following antiderivatives (indefinite integrals) and definite integrals.  Remember that antiderivatives need a ``$+C$''.

% problems from  Fall 2022 Derivative-v2.pdf ("retake 1")
\begin{subproblems}
\item $\ds \int_0^1 4e^x+\cos(x) \; dx$
\vfill

\item $\ds \int x+x\sin(x^2+1)\; dx$
\vfill

\item $\ds \int \frac{7-x+x^4}{x^2}  \; dx$
\vfill

\newpage
\item $\ds \int \frac{1+\sec^2(t)}{t+\tan(t)} \; dt$
\vfill

\item $\ds \int \frac{\cos(\arctan(x))}{1+x^2}\; dx$
\vfill

\item $\ds \int x(x+1)^5\; dx$
\vfill
\end{subproblems}
