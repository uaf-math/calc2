\vfill

\newpage
\item   $\ds f(x) =\ln(\tan(2x)+\sec(2x))$
\vfill


\item  $\ds h(x) = \left( x+ \sin(x^2+1) \right)^{-2}$
\vfill

\item  $\ds h(x) = \frac{1}{5x}+ \arctan(x^3)$

\vfill
\end{subproblems}

\problem{12}  Compute the following antiderivatives (indefinite integrals) and definite integrals.  Remember that antiderivatives need a ``$+C$''.

% problems from  Fall 2022 Derivative-v2.pdf ("retake 1")
\begin{subproblems}
\item $\ds \int_{-1}^1 x(2-x) \; dx$
\vfill

\item $\ds \int \sin(\pi x)+\frac{2}{3x}\; dx$
\vfill

\item $\ds \int \frac{x}{\sqrt{2+x^2}}  \; dx$
\vfill

\newpage
\item $\ds \int_0^{\pi/2} \cos(x)(\sin(x) +1)^{3} \; dx$
\vfill

\item $\ds \int \frac{e^x}{1+e^{2x}}\; dx$
\vfill

\item $\ds \int \frac{x}{(x+1)^2}\; dx$
\vfill
\end{subproblems}
