30 minutes maximum.  No aids (book, calculator, etc.) are permitted.  Show all work and use proper notation for full credit.  Answers should be in reasonably-simplified form.  25 points possible.

\problem{5}  Find the area of the region between $y=\sin x$ and $y=\cos x$ on the interval $[0,\pi/2]$.  (\textsl{Hint: Draw a careful sketch first!  You may use symmetry if you want.})
\vfill

\problem{15}

\begin{subproblems}
\item Sketch the region bounded by $y=x^2$, $y=0$, and $x=1$.  Then sketch the solid of revolution formed by rotating the region around the $x$-axis.  \emph{Please make your sketches adequately large and clear!}
\vfill

\newpage
\item Find the volume of the solid which you sketched in part \textbf{a}.  (\textsl{Hint: Use discs or washers.})
\vfill

\item Find the volume of the solid formed by revolving the region in part \textbf{a}  around the $y$-axis.  (\textsl{Hint: Sketch the solid.  Use discs or washers.})
\vfill
\end{subproblems}

\newpage
\problem{5}  A solid has a base which is the unit circle in the $x,y$ plane, and each cross-section parallel to the $y$-axis is a square.  Find the volume.
\vfill

\newpage
\nameprob{EC}{1} (\emph{Extra Credit})  Give the correct value of the definite integral:
	$$\int_{-1}^1 \sqrt{1-x^2}\,dx.$$
(\emph{Hint.}  There is no requirement to use the fundamental theorem of calculus.  What is sought is the correct answer, \emph{with some justification}, which might be a sketch.)
\vfill
