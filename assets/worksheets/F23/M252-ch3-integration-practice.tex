\vfill

\vfill
\item $\displaystyle{\int  \frac{\ln(\ln(x))}{x \ln(x)}\: dx}$

\item $\displaystyle{\int  \sin^6(x) \cos^3(x)\: dx}$
\vfill
\item $\displaystyle{\int  x\sin(7x) \: dx}$
\vfill
\end{enumerate}
