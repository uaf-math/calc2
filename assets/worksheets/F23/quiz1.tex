30 minutes maximum.  24 points possible; each part is worth 2 points.  No aids (book, notes, calculator, phone, etc.) are permitted.  Show all work and use proper notation for full credit.  Answers should be in reasonably-simplified form.

\problem{12}  Compute the derivatives of the following functions.

% problems from  Spring 2021 Derivative-v2.pdf ("retake 1")
\begin{subproblems}
\item  $\ds f(x)=e^2 x^{1/2} + 2 e^x + \sqrt{9}$
\vfill

\item   $\ds f(x) = \ln\left(\cos(x^3) - 4 x^7\right)$
\vfill

\item   $\ds h(x)=\sin(k x^2 - 5)$ where $k$ is a constant
\vfill

\newpage
\item   $\ds f(x) =\sec\left(x e^x\right)$
\vfill

\item  $\ds y = \frac{\cos(2x)}{x^5 + \pi}$
\vfill

\item  Find $\ds \frac{dy}{dx}$ if $\ds e^y \cos(x)=xy + 1$.  You must solve for $\ds \frac{dy}{dx}$.
\vfill
\end{subproblems}


\newpage
\problem{12}  Compute the following antiderivatives (indefinite integrals) and definite integrals.  Remember that antiderivatives need a ``$+C$''.

% problems from  Spring 2021 Derivative-v2.pdf ("retake 1")
\begin{subproblems}
\item $\ds \int \frac{(1+x)^2}{2x} \; dx$
\vfill

\item $\ds \int (x-1)\, e^{\left((x-1)^2\right)}\; dx$
\vfill

\item $\ds \int_0^\pi 5 e^x + 3 \sin(x)  \; dx$
\vfill

\newpage
\item $\ds \int x\sqrt{x+5} \; dx$
\vfill

\item $\ds \int \frac{\cos(\ln x)}{x}\; dx$
\vfill

\item $\ds \int \frac{\sec^2(x)}{\tan^2(x)}\; dx$
\vfill
\end{subproblems}
